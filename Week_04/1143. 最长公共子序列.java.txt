class Solution {
    public int longestCommonSubsequence(String text1, String text2) {

    int m=text1.length();
    int n=text2.length();
    //定义一个m*n的数组；
    int[][] dp=new int[m+1][n+1];
    for(int i=1;i<m+1;i++){
        for(int j=1;j<n+1;j++){
        //判断最后一个字符是否相等，相等就都-1，然后结果加一
        if(text1.charAt(i-1)==text2.charAt(j-1)){
          dp[i][j]=dp[i-1][j-1]+1;
        }else{
         //最后一个字符不相等
         dp[i][j]=Math.max(dp[i-1][j],dp[i][j-1]);
        }
    }
 }
    return dp[m][n];
    }
}
